    <script>
        // Rates configuration
        const PEAK_RATE = 0.287; // £0.287 per kWh
        const OFF_PEAK_RATE = 0.090; // £0.090 per kWh
        const STANDING_CHARGE_RATE = 0.60; // £0.60 per day

        let rawData = [];
        let currentView = 'daily';
        let mainChart = null;

        function handleFileUpload(event) {
            const file = event.target.files[0];
            if (!file) return;

            document.getElementById('upload-text').innerText = file.name;

            const reader = new FileReader();
            reader.onload = function(e) {
                const text = e.target.result;
                parseCSV(text);
            };
            reader.readAsText(file);
        }

        function parseCSV(text) {
            const lines = text.split(/\r?\n/);
            if (lines.length < 2) return;

            // Simple header matching to find column indexes
            const headers = lines[0].split(',').map(h => h.trim().replace(/['"]/g, ''));
            const kwhIndex = headers.indexOf('kWh');
            const dateIndex = headers.indexOf('dateTime');

            if (kwhIndex === -1 || dateIndex === -1) {
                alert("Error: Invalid CSV format. Must contain 'kWh' and 'dateTime' columns.");
                return;
            }

            rawData = [];

            for (let i = 1; i < lines.length; i++) {
                if (!lines[i].trim()) continue;
                const row = lines[i].split(',');
                if (row.length <= Math.max(kwhIndex, dateIndex)) continue;

                const kwhVal = parseFloat(row[kwhIndex]);
                const dateStr = row[dateIndex].replace(/['"]/g, '').trim();
                
                if (isNaN(kwhVal) || !dateStr) continue;

                const dateObj = new Date(dateStr);
                if (isNaN(dateObj.getTime())) continue;

                // Determine if peak or off-peak based on local/UTC hour
                const hour = dateObj.getHours(); 
                const period = (hour >= 0 && hour < 6) ? 'offpeak' : 'peak';

                rawData.push({
                    date: dateObj,
                    kwh: kwhVal,
                    period: period
                });
            }

            if (rawData.length > 0) {
                document.getElementById('no-data-alert').classList.add('hidden');
                document.getElementById('dashboard-content').classList.remove('hidden');
                processAndRenderView();
            }
        }

        // Helper to count unique dates in an array of records
        function getUniqueDaysCount(records) {
            const days = new Set();
            records.forEach(r => {
                const yyyy = r.date.getFullYear();
                const mm = String(r.date.getMonth() + 1).padStart(2, '0');
                const dd = String(r.date.getDate()).padStart(2, '0');
                days.add(`${yyyy}-${mm}-${dd}`);
            });
            return days.size;
        }

        function switchView(view) {
            currentView = view;
            ['halfhourly', 'hourly', 'daily', 'weekly', 'monthly'].forEach(v => {
                document.getElementById(`tab-${v}`).classList.remove('tab-active');
            });
            document.getElementById(`tab-${view}`).classList.add('tab-active');
            processAndRenderView();
        }

        function getGroupKey(date, view) {
            const yyyy = date.getFullYear();
            const mm = String(date.getMonth() + 1).padStart(2, '0');
            const dd = String(date.getDate()).padStart(2, '0');
            
            if (view === 'halfhourly') {
                const mins = date.getMinutes() < 30 ? '00' : '30';
                return `${yyyy}-${mm}-${dd} ${String(date.getHours()).padStart(2, '0')}:${mins}`;
            } else if (view === 'hourly') {
                return `${yyyy}-${mm}-${dd} ${String(date.getHours()).padStart(2, '0')}:00`;
            } else if (view === 'daily') {
                return `${yyyy}-${mm}-${dd}`;
            } else if (view === 'weekly') {
                const d = new Date(date);
                const day = d.getDay();
                const diff = d.getDate() - day + (day === 0 ? -6 : 1); 
                const monday = new Date(d.setDate(diff));
                return `Week of ${monday.getDate()} ${monday.toLocaleString('default', { month: 'short' })} ${monday.getFullYear()}`;
            } else if (view === 'monthly') {
                return date.toLocaleString('default', { month: 'long', year: 'numeric' });
            }
        }

        function processAndRenderView() {
            const groups = {};

            rawData.forEach(d => {
                const key = getGroupKey(d.date, currentView);
                const dateKey = `${d.date.getFullYear()}-${String(d.date.getMonth() + 1).padStart(2, '0')}-${String(d.date.getDate()).padStart(2, '0')}`;
                
                if (!groups[key]) {
                    groups[key] = { 
                        peakKwh: 0, 
                        offpeakKwh: 0,
                        uniqueDays: new Set()
                    };
                }
                
                groups[key].uniqueDays.add(dateKey);
                if (d.period === 'peak') groups[key].peakKwh += d.kwh;
                else groups[key].offpeakKwh += d.kwh;
            });

            // Sort chronically for the Chart (oldest to newest is natural left-to-right)
            const labelsForChart = Object.keys(groups);
            if (['halfhourly', 'hourly', 'daily'].includes(currentView)) {
                labelsForChart.sort();
            } else if (currentView === 'monthly') {
                // Ensure monthly order is chronological
                labelsForChart.sort((a, b) => new Date(a) - new Date(b));
            }

            const peakData = [];
            const offpeakData = [];
            
            labelsForChart.forEach(label => {
                const item = groups[label];
                peakData.push(item.peakKwh);
                offpeakData.push(item.offpeakKwh);
            });

            // Calculate metrics for dynamic cards based on the processed view
            updateDynamicStats(groups);

            // For the Table: Reverse so today / newest is at the top
            const labelsForTable = [...labelsForChart].reverse();
            const tableBody = document.getElementById('table-body');
            tableBody.innerHTML = '';

            document.getElementById('table-time-col').innerText = 
                currentView === 'halfhourly' ? 'Half-Hour' :
                currentView === 'hourly' ? 'Hour' :
                currentView === 'daily' ? 'Date' : 
                currentView === 'weekly' ? 'Week' : 'Month';
                
            document.getElementById('table-heading').innerText = `${currentView === 'halfhourly' ? 'Half-Hourly' : currentView.charAt(0).toUpperCase() + currentView.slice(1)} Bill Breakdown (Newest First)`;

            // Distribute standing charge fairly across fractional daily periods to prevent overcharging 
            let standingFraction = 1;
            if (currentView === 'hourly') standingFraction = 1 / 24;
            else if (currentView === 'halfhourly') standingFraction = 1 / 48;

            labelsForTable.forEach(label => {
                const item = groups[label];
                const pCost = item.peakKwh * PEAK_RATE;
                const opCost = item.offpeakKwh * OFF_PEAK_RATE;
                const energyCost = pCost + opCost;
                
                const standingDays = item.uniqueDays.size;
                const standingCost = standingDays * STANDING_CHARGE_RATE * standingFraction;
                const totalCost = energyCost + standingCost;

                // Add calculations for new columns
                const totalKwh = item.peakKwh + item.offpeakKwh;
                const peakPct = totalKwh > 0 ? (item.peakKwh / totalKwh) * 100 : 0;
                const offpeakPct = totalKwh > 0 ? (item.offpeakKwh / totalKwh) * 100 : 0;

                const row = document.createElement('tr');
                row.className = "hover:bg-neutral-800/40 transition duration-150";
                row.innerHTML = `
                    <td class="p-4 font-semibold text-neutral-100">
                        ${label}
                        <div class="text-xs text-neutral-500 font-normal">${standingDays} day${standingDays > 1 ? 's' : ''} recorded</div>
                    </td>
                    <td class="p-4 text-emerald-400 font-medium">${item.offpeakKwh.toFixed(2)} kWh</td>
                    <td class="p-4 text-blue-400 font-medium">${item.peakKwh.toFixed(2)} kWh</td>
                    <!-- Column: Total Usage -->
                    <td class="p-4 text-neutral-300 font-bold">${totalKwh.toFixed(2)} kWh</td>
                    <!-- Column: Percentage Split (Off-Peak % / Peak %) -->
                    <td class="p-4 font-medium text-neutral-400">
                        <span class="text-emerald-400">${offpeakPct.toFixed(0)}%</span> / 
                        <span class="text-blue-400">${peakPct.toFixed(0)}%</span>
                    </td>
                    <td class="p-4 text-neutral-400">£${energyCost.toFixed(2)}</td>
                    <td class="p-4 text-amber-600">£${standingCost.toFixed(3)}</td>
                    <td class="p-4 font-bold text-neutral-100">£${totalCost.toFixed(2)}</td>
                `;
                tableBody.appendChild(row);
            });

            if (mainChart) {
                mainChart.destroy();
            }

            const ctx = document.getElementById('usageChart').getContext('2d');
            
            // Adjust opacity dynamically: Bar charts need high opacity so they aren't "dim"
            const isMonthly = currentView === 'monthly';
            const offpeakBgColor = isMonthly ? 'rgba(52, 211, 153, 0.7)' : 'rgba(52, 211, 153, 0.05)';
            const peakBgColor = isMonthly ? 'rgba(96, 165, 250, 0.7)' : 'rgba(96, 165, 250, 0.05)';

            mainChart = new Chart(ctx, {
                type: isMonthly ? 'bar' : 'line',
                data: {
                    labels: labelsForChart,
                    datasets: [
                        {
                            label: 'Off-Peak Usage (kWh)',
                            data: offpeakData,
                            borderColor: '#34d399',
                            backgroundColor: offpeakBgColor,
                            fill: true,
                            tension: 0.15
                        },
                        {
                            label: 'Peak Usage (kWh)',
                            data: peakData,
                            borderColor: '#60a5fa',
                            backgroundColor: peakBgColor,
                            fill: true,
                            tension: 0.15
                        }
                    ]
                },
                options: {
                    responsive: true,
                    maintainAspectRatio: false,
                    scales: {
                        x: {
                            grid: { color: '#262626' },
                            ticks: { color: '#a3a3a3' }
                        },
                        y: {
                            beginAtZero: true,
                            grid: { color: '#262626' },
                            ticks: { color: '#a3a3a3' },
                            title: { display: true, text: 'Energy Consumption (kWh)', color: '#d4d4d4', font: { weight: 'bold' } }
                        }
                    },
                    plugins: {
                        legend: { 
                            position: 'top',
                            labels: { color: '#d4d4d4' }
                        }
                    }
                }
            });
        }

        function updateDynamicStats(groups) {
            let totalKwh = 0;
            let totalCost = 0;
            let totalStanding = 0;
            const periodKeys = Object.keys(groups);
            const periodCount = periodKeys.length;

            let standingFraction = 1;
            if (currentView === 'hourly') standingFraction = 1 / 24;
            else if (currentView === 'halfhourly') standingFraction = 1 / 48;

            periodKeys.forEach(key => {
                const item = groups[key];
                totalKwh += (item.peakKwh + item.offpeakKwh);
                
                const pCost = item.peakKwh * PEAK_RATE;
                const opCost = item.offpeakKwh * OFF_PEAK_RATE;
                const standingDays = item.uniqueDays.size;
                const standingCost = standingDays * STANDING_CHARGE_RATE * standingFraction;

                totalCost += (pCost + opCost + standingCost);
                totalStanding += standingCost;
            });

            const uniqueDays = getUniqueDaysCount(rawData);

            // Establish view label text values
            let periodLabelSingular = 'day';
            let periodLabelPlural = 'days';
            if (currentView === 'halfhourly') {
                periodLabelSingular = 'half-hour';
                periodLabelPlural = 'half-hours';
            } else if (currentView === 'hourly') {
                periodLabelSingular = 'hour';
                periodLabelPlural = 'hours';
            } else if (currentView === 'weekly') {
                periodLabelSingular = 'week';
                periodLabelPlural = 'weeks';
            } else if (currentView === 'monthly') {
                periodLabelSingular = 'month';
                periodLabelPlural = 'months';
            }

            const activePeriodLabel = periodCount === 1 ? periodLabelSingular : periodLabelPlural;

            // Averages
            const avgKwh = periodCount > 0 ? (totalKwh / periodCount) : 0;
            const avgCost = periodCount > 0 ? (totalCost / periodCount) : 0;
            const avgStanding = periodCount > 0 ? (totalStanding / periodCount) : 0;

            // Card 1: Dynamic Usage Card
            document.getElementById('lbl-total-kwh').innerText = `Avg. Usage / ${periodLabelSingular}`;
            document.getElementById('stat-total-kwh').innerHTML = `${avgKwh.toLocaleString(undefined, {minimumFractionDigits: 2, maximumFractionDigits: 2})} <span class="text-sm font-medium text-neutral-500">kWh</span>`;
            document.getElementById('sub-total-kwh').innerText = `Total: ${totalKwh.toLocaleString(undefined, {minimumFractionDigits: 2, maximumFractionDigits: 2})} kWh overall`;

            // Card 2: Dynamic Cost Card
            document.getElementById('lbl-total-cost').innerText = `Avg. Cost / ${periodLabelSingular}`;
            document.getElementById('stat-total-cost').innerText = `£${avgCost.toLocaleString(undefined, {minimumFractionDigits: 2, maximumFractionDigits: 2})}`;
            document.getElementById('sub-total-cost').innerText = `Total: £${totalCost.toLocaleString(undefined, {minimumFractionDigits: 2, maximumFractionDigits: 2})} overall`;

            // Card 3: Dynamic Standing Charge Card
            document.getElementById('lbl-standing-total').innerText = `Avg. Standing / ${periodLabelSingular}`;
            document.getElementById('stat-standing-total').innerText = `£${avgStanding.toLocaleString(undefined, {minimumFractionDigits: 2, maximumFractionDigits: 4})}`;
            document.getElementById('sub-standing-total').innerText = `Total: £${totalStanding.toLocaleString(undefined, {minimumFractionDigits: 2, maximumFractionDigits: 2})} overall`;

            // Card 4: Dynamic Span Card
            document.getElementById('lbl-days-span').innerText = `Data Span (${activePeriodLabel})`;
            document.getElementById('stat-days-span').innerText = `${periodCount} ${activePeriodLabel}`;
            document.getElementById('sub-days-span').innerText = `${uniqueDays} active logging day${uniqueDays > 1 ? 's' : ''}`;
        }
    </script>
